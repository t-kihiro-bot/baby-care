2
メニュー「拡張機能」→「Apps Script」を開き、
以下のコードを全て貼り付けて保存します。
function doGet(e){ var p=e.parameter, s=getSheet(); if(p.action==='add'){ var d=new Date(Number(p.ts)); var ml=p.ml?Number(p.ml):0; s.appendRow([p.id,p.type,p.cat,Number(p.ts), Utilities.formatDate(d,'Asia/Tokyo','yyyy/MM/dd HH:mm'),ml]); return ok(); } if(p.action==='del'){ var v=s.getDataRange().getValues(); for(var i=v.length-1;i>=1;i--){ if(String(v[i][0])===String(p.id)){s.deleteRow(i+1);break;} } return ok(); } if(p.action==='addgrowth'){ var gs=getGrowthSheet(); var d2=new Date(Number(p.ts)); gs.appendRow([p.id,Number(p.ts),p.weight||'',p.height||'', Utilities.formatDate(d2,'Asia/Tokyo','yyyy/MM/dd')]); return ok(); } if(p.action==='delgrowth'){ var gs=getGrowthSheet(),gv=gs.getDataRange().getValues(); for(var i=gv.length-1;i>=1;i--){ if(String(gv[i][0])===String(p.id)){gs.deleteRow(i+1);break;} } return ok(); } if(p.action==='readgrowth'){ var gs=getGrowthSheet(),gv=gs.getDataRange().getValues(),out=[]; for(var i=1;i<gv.length;i++){ if(!gv[i][0])continue; var rec={id:String(gv[i][0]),ts:Number(gv[i][1])}; if(gv[i][2]!=='')rec.weight=Number(gv[i][2]); if(gv[i][3]!=='')rec.height=Number(gv[i][3]); out.push(rec); } out.sort(function(a,b){return a.ts-b.ts;}); return json(out); } if(p.action==='setconfig'){ var cs=getConfigSheet(),cv=cs.getDataRange().getValues(); for(var i=1;i<cv.length;i++){ if(cv[i][0]===p.key){cs.getRange(i+1,2).setValue(p.value);return ok();} } cs.appendRow([p.key,p.value]); return ok(); } if(p.action==='getconfig'){ var cs=getConfigSheet(),cv=cs.getDataRange().getValues(),cfg={}; for(var i=1;i<cv.length;i++){ if(!cv[i][0]) continue; var val=cv[i][1]; if(val instanceof Date){val=Utilities.formatDate(val,'Asia/Tokyo','yyyy-MM-dd');} else{val=String(val);} cfg[cv[i][0]]=val; } return json(cfg); } var v=s.getDataRange().getValues(),out=[]; for(var i=1;i<v.length;i++){ if(!v[i][0])continue; var rec={id:String(v[i][0]),type:v[i][1],cat:v[i][2],ts:Number(v[i][3])}; if(v[i][5])rec.ml=Number(v[i][5]); out.push(rec); } out.sort(function(a,b){return b.ts-a.ts;}); return json(out); } function getSheet(){ var ss=SpreadsheetApp.getActiveSpreadsheet(); var s=ss.getSheetByName('records'); if(!s){s=ss.getActiveSheet();s.setName('records'); s.appendRow(['id','type','cat','ts','日時','ml']);} return s; } function getGrowthSheet(){ var ss=SpreadsheetApp.getActiveSpreadsheet(); var s=ss.getSheetByName('growth'); if(!s){s=ss.insertSheet('growth');s.appendRow(['id','ts','weight','height','日付']);} return s; } function getConfigSheet(){ var ss=SpreadsheetApp.getActiveSpreadsheet(); var s=ss.getSheetByName('config'); if(!s){s=ss.insertSheet('config');s.appendRow(['key','value']);} return s; } function ok(){return json({ok:true});} function json(d){ return ContentService.createTextOutput(JSON.stringify(d)) .setMimeType(ContentService.MimeType.JSON); }
